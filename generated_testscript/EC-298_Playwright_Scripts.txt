// Playwright script to test login and inventory access

// Step 1: Navigate to the application URL
await page.goto('https://www.saucedemo.com/');

// Step 2: Verify the visibility of the login fields
await expect(page.locator('#user-name')).toBeVisible();
await expect(page.locator('#password')).toBeVisible();
await expect(page.locator('.btn_action')).toBeVisible();

// Step 3: Enter username and password
await page.fill('#user-name', 'standard_user');
await page.fill('#password', 'sauce');

// Step 4: Click the Login button
await page.click('.btn_action');

// Step 5: Verify redirection to inventory page
await expect(page).toHaveURL(/.*inventory.html/);

// Step 6: Dismiss login popup if present
const loginPopup = page.locator('.login-popup');
if (await loginPopup.isVisible()) {
    await loginPopup.click('.dismiss-button');
}

// Step 7: Verify product list is visible
await expect(page.locator('.inventory_list')).toBeVisible();
